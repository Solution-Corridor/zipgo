CREATE TABLE dollar_prices (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    price DECIMAL(10,2) NOT NULL,
    created_at TIMESTAMP NULL DEFAULT NULL,
    updated_at TIMESTAMP NULL DEFAULT NULL
);

CREATE TABLE `mk_product_reviews` (
    `id` BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    `product_id` BIGINT UNSIGNED NOT NULL,
    `user_id` BIGINT UNSIGNED NULL,                    -- NULL if guest review
    `reviewer_name` VARCHAR(100) NULL,                 -- For guest users
    `rating` TINYINT UNSIGNED NOT NULL CHECK (rating BETWEEN 1 AND 5),
    `review` TEXT NULL,                                -- Feedback text
    `approved` TINYINT(1) DEFAULT 1,                   -- 1 = approved, 0 = pending
    `created_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
